04 — Arrays
push, map, filter, find, reduce, join
Try it → 
// Array creation, push, map, filter, find, reduce
main() I {
    a = [1 2 3 4 5]
    p("len = {a.len()}")
    p("first = {a[0]}, last = {a[4]}")

    a.push(6)
    p("after push: len = {a.len()}")

    doubled = a.map(|x:I| I { x * 2 })
    p("doubled[0] = {doubled[0]}, doubled[5] = {doubled[5]}")

    evens = a.filter(|x:I| B { x % 2 == 0 })
    p("evens len = {evens.len()}")

    found = a.find(|x:I| B { x > 4 })
    p("first > 4: {found!}")

    total = a.reduce(0 |acc:I x:I| I { acc + x })
    p("sum = {total}")

    joined = a.map(|x:I| S { str(x) }).join(", ")
    p("joined: {joined}")

    has3 = a.contains(3)
    p("contains 3: {has3}")

    rev = [10 20 30].reverse()
    p("reversed[0] = {rev[0]}")

    0
}

Concurrency

Threads, channels, mutexes, worker pools, and parallel algorithms — all built in, no libraries needed.

32 — Channels
Send and receive values between threads
Try it → 
// Send/receive values between threads via channels
sender(tx Sender<I>) I {
    tx.send(100)
    tx.send(200)
    tx.send(300)
    0
}

main() I {
    let (tx, rx) = channel<I>()
    h = spawn sender(tx)
    a = rx.recv()
    b = rx.recv()
    c = rx.recv()
    h.join()
    p("received: {a}, {b}, {c}")
    p("sum: {a + b + c}")
    0
}

43 — FizzBuzz
Classic problem, idiomatic Sans
Try it → 
fizzbuzz(n:I) S {
  fb = n % 15 == 0
  f  = n % 3 == 0
  b  = n % 5 == 0
  fb ? "FizzBuzz" : f ? "Fizz" : b ? "Buzz" : str(n)
}

main() I {
  i := 1
  while i <= 30 {
    p(fizzbuzz(i))
    i += 1
  }
  0
}
