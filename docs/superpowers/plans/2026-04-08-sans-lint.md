# `sans lint` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `sans lint` command that runs static analysis (parse + typeck) without building, catching common code issues with configurable severity.

**Architecture:** New `compiler/lint.sans` module runs after typeck on the typed AST. CLI dispatches `lint` subcommand in `main.sans`, reusing the existing parse/typeck pipeline but stopping before IR/codegen. Diagnostics flow through the existing `DIAG_WARN`/`DIAG_ERROR` system in typeck.sans.

**Tech Stack:** Sans (self-hosted compiler), LLVM 17

**Spec:** `docs/superpowers/specs/2026-04-08-sans-lint-design.md`

// Rule IDs

g LINT_UNREACHABLE_CODE = "unreachable-code"
g LINT_EMPTY_CATCH = "empty-catch"
g LINT_SHADOWED_VARS = "shadowed-vars"
g LINT_UNNECESSARY_MUT = "unnecessary-mut"

// Severities
g LINT_SEV_OFF = 0
g LINT_SEV_WARN = 1
g LINT_SEV_ERROR = 2

lint_init_config() I {
  if lint_rule_severity == 0 {
    lint_rule_severity = M()
    // Defaults: all warn
    mset(lint_rule_severity, LINT_UNUSED_IMPORTS, LINT_SEV_WARN)
    mset(lint_rule_severity, LINT_UNREACHABLE_CODE, LINT_SEV_WARN)
    mset(lint_rule_severity, LINT_EMPTY_CATCH, LINT_SEV_WARN)
    mset(lint_rule_severity, LINT_SHADOWED_VARS, LINT_SEV_WARN)
    mset(lint_rule_severity, LINT_UNNECESSARY_MUT, LINT_SEV_WARN)
  }
  0
}

lint_get_severity(rule:S) I {
  if lint_rule_severity == 0 { return LINT_SEV_WARN }
  if mhas(lint_rule_severity, rule) { return mget(lint_rule_severity, rule) }
  LINT_SEV_WARN
}

// ------ Lint Pass (placeholder — rules added in later tasks) -------------------------------------

g lint_has_error = 0

lint_emit(node:I msg:S rule:S) I {
  sev = lint_get_severity(rule)
  if sev == LINT_SEV_OFF { return 0 }
  tagged_msg = msg + " [" + rule + "]"
  if sev == LINT_SEV_ERROR {
    tc_error_at(node, tagged_msg)
    lint_has_error = 1
  } else {
    tc_warn_at(node, tagged_msg)
  }
  0
}

lint_check(program:I) I {
  // Rules will be added in subsequent tasks
  0
}

// ------ Lint Entry Points ------------------------------------------------------------------------

lint_file(path:S) I {
  src = fr(path)
  if slen(src) == 0 {
    p("sans lint: cannot read file: " + path)
    return 1
  }

  // Parse
  program = parse(src)

  // Resolve imports
  modules = resolve_imports(path)

  // Type-check imported modules
  mod_exports = M()
  i := 0
  while i < modules.len() {
    mod = modules.get(i)
    mod_n = rm_name(mod)
    mod_prog = rm_program(mod)
    mod_path = rm_path(mod)
    tc_current_file = mod_path
    tc_current_source = fr(mod_path)
    tc_has_source = 1
    exports = check_module(mod_prog, mod_exports)
    merge_reexports(exports, mod_prog, mod_exports, mod_n)
    mset(mod_exports, mod_n, exports)
    mod_alias = rm_alias(mod)
    if mod_alias != 0 { mset(mod_exports, mod_alias, exports) }
    i = i + 1
  }

  // Type-check entry file with warnings enabled
  tc_current_file = path
  tc_current_source = src
  tc_has_source = 1
  tc_warn_enabled = 1
  check_module(program, mod_exports)
  tc_warn_enabled = 0

  // Run lint pass
  lint_check(program)

  // Emit diagnostics (don't exit on error — we handle exit ourselves)
  tc_emit_diags()
  0
}

lint_dir(dir:S) I {
  files = ls(dir)
  i := 0
  while i < files.len() {
    fname = files.get(i)
    if fname.ends_with(".sans") == 1 {
      path = dir + "/" + fname
      lint_file(path)
    }
    // Recurse into subdirectories
    if fname != "." && fname != ".." {
      subpath = dir + "/" + fname
      if is_dir(subpath) == 1 {
        lint_dir(subpath)
      }
    }
    i = i + 1
  }
  0
}

lint_main(target:S argv:I argc:I) I {
  lint_init_config()

  // Parse CLI flags: --error=<rule>, --quiet
  quiet := 0
  fi := 3
  while fi < argc {
    arg = argv.get(fi)
    if arg == "--quiet" {
      quiet = 1
    }
    if arg.starts_with("--error=") == 1 {
      rule = arg.slice(8, slen(arg))
      if mhas(lint_rule_severity, rule) {
        mset(lint_rule_severity, rule, LINT_SEV_ERROR)
      } else {
        p("sans lint: unknown rule '" + rule + "'")
        exit(1)
      }
    }
    fi += 1
  }

  // Parse sans.json lint config (CLI overrides already applied above)
  lint_load_config()

  lint_has_error = 0

  if is_dir(target) == 1 {
    lint_dir(target)
  } else {
    lint_file(target)
  }

  if lint_has_error == 1 { exit(1) }
  0
}

lint_load_config() I {
  // Try reading sans.json for lint config
  if fexists("sans.json") == 0 { return 0 }
  src = fr("sans.json")
  if slen(src) == 0 { return 0 }
  j = json_parse(src)
  if j == 0 { return 0 }
  lint_obj = json_get(j, "lint")
  if lint_obj == 0 { return 0 }
  keys = json_keys(lint_obj)
  if keys == 0 { return 0 }
  ki := 0
  while ki < keys.len() {
    k = keys.get(ki)
    // Only set if not already overridden by CLI
    v = json_get_string(lint_obj, k)
    if mhas(lint_rule_severity, k) {
      if v == "error" { mset(lint_rule_severity, k, LINT_SEV_ERROR) }
      if v == "warn" { mset(lint_rule_severity, k, LINT_SEV_WARN) }
      if v == "off" { mset(lint_rule_severity, k, LINT_SEV_OFF) }
    }
    ki += 1
  }
  0
}

- [ ] **Step 3: Implement `lint_unused_imports` in `lint.sans`**

The rule walks the program's import list, then checks whether any function/type from that module was used. The typeck pass has already resolved imports and built `mod_exports`. We need to track which module names are actually referenced in the code.

Add to `lint.sans`:

```sans
// ------ Rule: unused-imports -------------------------------------------------------------------

lint_unused_imports(program:I) I {
  if lint_get_severity(LINT_UNUSED_IMPORTS) == LINT_SEV_OFF { return 0 }

  imports_arr = load64(program)  // offset 0: imports array
  num_imports = imports_arr.len()
  if num_imports == 0 { return 0 }

  // Collect all function/method call names and type references in the program
  used_names = M()
  fns = load64(program + 16)  // functions array
  fi := 0
  while fi < fns.len() {
    fn_node = fns.get(fi)
    fn_body = load64(fn_node + 40)  // body statements ptr
    if fn_body != 0 {
      fn_body_len = load64(fn_body + 8)
      fn_body_p = load64(fn_body)
      lint_collect_names(fn_body_p, fn_body_len, used_names)
    }
    fi += 1
  }

  // Also collect from impl blocks
  impls = load64(program + 48)  // impls array
  ii := 0
  while ii < impls.len() {
    impl_node = impls.get(ii)
    impl_methods = load64(impl_node + 16)  // methods array
    mi := 0
    while mi < impl_methods.len() {
      m_node = impl_methods.get(mi)
      m_body = load64(m_node + 40)
      if m_body != 0 {
        m_body_len = load64(m_body + 8)
        m_body_p = load64(m_body)
        lint_collect_names(m_body_p, m_body_len, used_names)
      }
      mi += 1
    }
    ii += 1
  }

  // Check each import — if no name from that module is used, warn
  // For now, check if the module name appears as a prefix in any call (mod__fn)
  // or if any function known to be from that module was called
  i := 0
  while i < num_imports {
    imp_node = imports_arr.get(i)
    imp_path = imp_get_path(imp_node)
    mod_name = mod_name_of(imp_path)
    prefix = mod_name + "__"
    // Check if any key in used_names starts with the module prefix
    found := 0
    ukeys = used_names.keys()
    uk := 0
    while uk < ukeys.len() {
      uname = ukeys.get(uk)
      if uname.starts_with(prefix) == 1 { found = 1 }
      // Also check direct name match (for re-exported functions)
      if uname == mod_name { found = 1 }
      uk += 1
    }
    if found == 0 {
      lint_emit(imp_node, "unused import '" + mod_name + "'", LINT_UNUSED_IMPORTS)
    }
    i += 1
  }
  0
}

// Collect all identifier/call names from statement list into names map
lint_collect_names(stmts_p:I count:I names:I) I {
  i := 0
  while i < count {
    stmt = load64(stmts_p + i * 8)
    lint_collect_names_stmt(stmt, names)
    i += 1
  }
  0
}

lint_collect_names_stmt(stmt:I names:I) I {
  tag = load64(stmt)

  if tag == ST_LET {
    value_expr = load64(stmt + 40)
    lint_collect_names_expr(value_expr, names)
    return 0
  }
  if tag == ST_ASSIGN {
    value_expr = load64(stmt + 24)
    lint_collect_names_expr(value_expr, names)
    return 0
  }
  if tag == ST_RETURN {
    value_expr = load64(stmt + 16)
    if value_expr != 0 { lint_collect_names_expr(value_expr, names) }
    return 0
  }
  if tag == ST_WHILE {
    cond = load64(stmt + 16)
    lint_collect_names_expr(cond, names)
    body_arr = load64(stmt + 24)
    if body_arr != 0 {
      lint_collect_names(load64(body_arr), load64(body_arr + 8), names)
    }
    return 0
  }
  if tag == ST_IF {
    cond = load64(stmt + 16)
    lint_collect_names_expr(cond, names)
    body_arr = load64(stmt + 24)
    if body_arr != 0 {
      lint_collect_names(load64(body_arr), load64(body_arr + 8), names)
    }
    // else branch
    else_arr = load64(stmt + 32)
    if else_arr != 0 {
      lint_collect_names(load64(else_arr), load64(else_arr + 8), names)
    }
    return 0
  }
  if tag == ST_EXPR {
    expr = load64(stmt + 16)
    if expr != 0 { lint_collect_names_expr(expr, names) }
    return 0
  }
  if tag == ST_FOR_IN {
    iter_expr = load64(stmt + 24)
    lint_collect_names_expr(iter_expr, names)
    body_arr = load64(stmt + 32)
    if body_arr != 0 {
      lint_collect_names(load64(body_arr), load64(body_arr + 8), names)
    }
    return 0
  }
  0
}

lint_collect_names_expr(expr:I names:I) I {
  if expr == 0 { return 0 }
  tag = load64(expr)

  if tag == EX_IDENT {
    name = load64(expr + 8)
    mset(names, name, 1)
    return 0
  }
  if tag == EX_CALL {
    callee = load64(expr + 8)
    mset(names, callee, 1)
    // Recurse into arguments
    args_arr = load64(expr + 16)
    if args_arr != 0 {
      ai := 0
      while ai < args_arr.len() {
        lint_collect_names_expr(args_arr.get(ai), names)
        ai += 1
      }
    }
    return 0
  }
  if tag == EX_METHOD_CALL {
    receiver = load64(expr + 8)
    lint_collect_names_expr(receiver, names)
    args_arr = load64(expr + 24)
    if args_arr != 0 {
      ai := 0
      while ai < args_arr.len() {
        lint_collect_names_expr(args_arr.get(ai), names)
        ai += 1
      }
    }
    return 0
  }
  if tag == EX_BINOP {
    lint_collect_names_expr(load64(expr + 8), names)
    lint_collect_names_expr(load64(expr + 24), names)
    return 0
  }
  if tag == EX_UNARYOP {
    lint_collect_names_expr(load64(expr + 16), names)
    return 0
  }
  if tag == EX_IF {
    lint_collect_names_expr(load64(expr + 8), names)   // cond
    lint_collect_names_expr(load64(expr + 24), names)  // then_expr
    lint_collect_names_expr(load64(expr + 40), names)  // else_expr
    return 0
  }
  if tag == EX_STRUCT_LIT {
    name = load64(expr + 8)
    mset(names, name, 1)
    return 0
  }
  if tag == EX_ENUM_VARIANT {
    name = load64(expr + 8)
    mset(names, name, 1)
    return 0
  }
  0
}
```

// ------ Rule: shadowed-vars --------------------------------------------------------------------

lint_shadowed_vars(program:I) I {
  if lint_get_severity(LINT_SHADOWED_VARS) == LINT_SEV_OFF { return 0 }

  fns = load64(program + 16)  // functions array
  fi := 0
  while fi < fns.len() {
    fn_node = fns.get(fi)
    fn_body = load64(fn_node + 40)
    if fn_body != 0 {
      // Collect parameter names as outer scope
      outer_scope = M()
      params = load64(fn_node + 16)  // params array
      if params != 0 {
        pi := 0
        while pi < params.len() {
          param = params.get(pi)
          param_name = load64(param + 8)  // param name at offset 8
          mset(outer_scope, param_name, fn_node)
          pi += 1
        }
      }
      fn_body_len = load64(fn_body + 8)
      fn_body_p = load64(fn_body)
      lint_shadow_stmts(fn_body_p, fn_body_len, outer_scope)
    }
    fi += 1
  }

  // Also check impl methods
  impls = load64(program + 48)
  ii := 0
  while ii < impls.len() {
    impl_node = impls.get(ii)
    impl_methods = load64(impl_node + 16)
    mi := 0
    while mi < impl_methods.len() {
      m_node = impl_methods.get(mi)
      m_body = load64(m_node + 40)
      if m_body != 0 {
        outer_scope = M()
        params = load64(m_node + 16)
        if params != 0 {
          pi := 0
          while pi < params.len() {
            param = params.get(pi)
            param_name = load64(param + 8)
            mset(outer_scope, param_name, m_node)
            pi += 1
          }
        }
        m_body_len = load64(m_body + 8)
        m_body_p = load64(m_body)
        lint_shadow_stmts(m_body_p, m_body_len, outer_scope)
      }
      mi += 1
    }
    ii += 1
  }
  0
}

// Walk statements tracking variable declarations per scope level
lint_shadow_stmts(stmts_p:I count:I outer_vars:I) I {
  // Track new declarations at this scope level
  current_vars = M()
  // Merge outer into current for sub-scope checking
  okeys = outer_vars.keys()
  oi := 0
  while oi < okeys.len() {
    mset(current_vars, okeys.get(oi), mget(outer_vars, okeys.get(oi)))
    oi += 1
  }

  i := 0
  while i < count {
    stmt = load64(stmts_p + i * 8)
    tag = load64(stmt)

    // Check variable declarations
    if tag == ST_LET {
      name = load64(stmt + 16)
      if mhas(outer_vars, name) {
        orig = mget(outer_vars, name)
        orig_line = node_line(orig)
        lint_emit(stmt, "variable '" + name + "' shadows outer declaration at line " + str(orig_line), LINT_SHADOWED_VARS)
      }
      mset(current_vars, name, stmt)
    }
    if tag == ST_ASSIGN {
      name = load64(stmt + 16)
      // Only flag if this is a NEW variable (not existing in any scope)
      // ST_ASSIGN to existing var is reassignment, not shadowing
      // But if it creates a new var in inner scope that shadows outer, flag it
      // Actually: bare assignment in Sans creates a new var if not found — shadowing is
      // really about inner scoped blocks (if/while) re-declaring
    }

    // Recurse into sub-blocks with current scope as outer
    if tag == ST_IF {
      body_arr = load64(stmt + 24)
      if body_arr != 0 {
        lint_shadow_stmts(load64(body_arr), load64(body_arr + 8), current_vars)
      }
      else_arr = load64(stmt + 32)
      if else_arr != 0 {
        lint_shadow_stmts(load64(else_arr), load64(else_arr + 8), current_vars)
      }
    }
    if tag == ST_WHILE {
      body_arr = load64(stmt + 24)
      if body_arr != 0 {
        lint_shadow_stmts(load64(body_arr), load64(body_arr + 8), current_vars)
      }
    }
    if tag == ST_FOR_IN {
      // Loop variable could shadow
      iter_name = load64(stmt + 16)
      if mhas(outer_vars, iter_name) || mhas(current_vars, iter_name) {
        orig = if mhas(outer_vars, iter_name) { mget(outer_vars, iter_name) } else { mget(current_vars, iter_name) }
        orig_line = node_line(orig)
        lint_emit(stmt, "variable '" + iter_name + "' shadows outer declaration at line " + str(orig_line), LINT_SHADOWED_VARS)
      }
      body_arr = load64(stmt + 32)
      if body_arr != 0 {
        inner = M()
        ckeys = current_vars.keys()
        ci := 0
        while ci < ckeys.len() {
          mset(inner, ckeys.get(ci), mget(current_vars, ckeys.get(ci)))
          ci += 1
        }
        mset(inner, iter_name, stmt)
        lint_shadow_stmts(load64(body_arr), load64(body_arr + 8), inner)
      }
    }
    i += 1
  }
  0
}

    fi += 1
  }

lint_check_mut_fn(stmts_p:I count:I) I {
  // Pass 1: collect all ST_LET with mutable=1
  mut_vars = M()  // Map<name, stmt_node>
  lint_collect_mut_decls(stmts_p, count, mut_vars)

  // Pass 2: collect all reassignment targets
  reassigned = M()  // Map<name, 1>
  lint_collect_reassignments(stmts_p, count, reassigned)

  // Pass 3: warn on mutable vars that were never reassigned
  mkeys = mut_vars.keys()
  mk := 0
  while mk < mkeys.len() {
    name = mkeys.get(mk)
    if mhas(reassigned, name) == 0 {
      // Skip variables starting with _ (intentionally unused/mutable)
      if char_at(name, 0) != 95 {
        lint_emit(mget(mut_vars, name), "variable '" + name + "' declared mutable (:=) but never reassigned — use '=' instead", LINT_UNNECESSARY_MUT)
      }
    }
    mk += 1
  }
  0
}

lint_collect_mut_decls(stmts_p:I count:I mut_vars:I) I {
  i := 0
  while i < count {
    stmt = load64(stmts_p + i * 8)
    tag = load64(stmt)
    if tag == ST_LET {
      mutable = load64(stmt + 24)
      if mutable == 1 {
        name = load64(stmt + 16)
        mset(mut_vars, name, stmt)
      }
    }
    // Recurse into sub-blocks
    if tag == ST_IF {
      body_arr = load64(stmt + 24)
      if body_arr != 0 { lint_collect_mut_decls(load64(body_arr), load64(body_arr + 8), mut_vars) }
      else_arr = load64(stmt + 32)
      if else_arr != 0 { lint_collect_mut_decls(load64(else_arr), load64(else_arr + 8), mut_vars) }
    }
    if tag == ST_WHILE {
      body_arr = load64(stmt + 24)
      if body_arr != 0 { lint_collect_mut_decls(load64(body_arr), load64(body_arr + 8), mut_vars) }
    }
    if tag == ST_FOR_IN {
      body_arr = load64(stmt + 32)
      if body_arr != 0 { lint_collect_mut_decls(load64(body_arr), load64(body_arr + 8), mut_vars) }
    }
    i += 1
  }
  0
}

lint_collect_reassignments(stmts_p:I count:I reassigned:I) I {
  i := 0
  while i < count {
    stmt = load64(stmts_p + i * 8)
    tag = load64(stmt)
    if tag == ST_ASSIGN {
      name = load64(stmt + 16)
      mset(reassigned, name, 1)
    }
    // Also catch compound assignments (+=, -=, etc.) — these show up as ST_ASSIGN
    // Recurse into sub-blocks
    if tag == ST_IF {
      body_arr = load64(stmt + 24)
      if body_arr != 0 { lint_collect_reassignments(load64(body_arr), load64(body_arr + 8), reassigned) }
      else_arr = load64(stmt + 32)
      if else_arr != 0 { lint_collect_reassignments(load64(else_arr), load64(else_arr + 8), reassigned) }
    }
    if tag == ST_WHILE {
      body_arr = load64(stmt + 24)
      if body_arr != 0 { lint_collect_reassignments(load64(body_arr), load64(body_arr + 8), reassigned) }
    }
    if tag == ST_FOR_IN {
      body_arr = load64(stmt + 32)
      if body_arr != 0 { lint_collect_reassignments(load64(body_arr), load64(body_arr + 8), reassigned) }
    }
    i += 1
  }
  0
}

- [ ] **Step 2: Implement `lint_empty_catch` in `lint.sans`**

The `?` operator desugars to `.unwrap()` method call at parse time, so we can't detect it in the AST. Instead, this rule detects when a function that returns `Result<T>` is called in a statement expression context (ST_EXPR) where the return value is discarded.

Add to `lint.sans`:

```sans
// ------ Rule: empty-catch (discarded Result) ---------------------------------------------------

lint_empty_catch(program:I) I {
  if lint_get_severity(LINT_EMPTY_CATCH) == LINT_SEV_OFF { return 0 }

  // Collect function names that return Result types
  // We check the AST for functions with Result return type annotations
  result_fns = M()
  fns = load64(program + 16)
  fi := 0
  while fi < fns.len() {
    fn_node = fns.get(fi)
    fn_name = load64(fn_node + 8)  // name at offset 8
    ret_type = load64(fn_node + 24)  // return type at offset 24
    if ret_type != 0 {
      // Check if return type is Result (type tag or string starts with "R<" or "Result<")
      ret_str = type_node_to_string(ret_type)
      if ret_str.starts_with("R<") == 1 || ret_str.starts_with("Result<") == 1 {
        mset(result_fns, fn_name, 1)
      }
    }
    fi += 1
  }

  // Walk all function bodies looking for ST_EXPR containing calls to result_fns
  fi = 0
  while fi < fns.len() {
    fn_node = fns.get(fi)
    fn_body = load64(fn_node + 40)
    if fn_body != 0 {
      lint_check_discarded_results(load64(fn_body), load64(fn_body + 8), result_fns)
    }
    fi += 1
  }

  impls = load64(program + 48)
  ii := 0
  while ii < impls.len() {
    impl_node = impls.get(ii)
    impl_methods = load64(impl_node + 16)
    mi := 0
    while mi < impl_methods.len() {
      m_node = impl_methods.get(mi)
      m_body = load64(m_node + 40)
      if m_body != 0 {
        lint_check_discarded_results(load64(m_body), load64(m_body + 8), result_fns)
      }
      mi += 1
    }
    ii += 1
  }
  0
}

// Helper to convert type AST node to string for Result detection
type_node_to_string(t:I) S {
  if t == 0 { return "" }
  // Type nodes store the type name string
  as_str(t)
}

lint_check_discarded_results(stmts_p:I count:I result_fns:I) I {
  i := 0
  while i < count {
    stmt = load64(stmts_p + i * 8)
    tag = load64(stmt)
    if tag == ST_EXPR {
      expr = load64(stmt + 16)
      if expr != 0 {
        etag = load64(expr)
        if etag == EX_CALL {
          callee = load64(expr + 8)
          if mhas(result_fns, callee) {
            lint_emit(stmt, "result of '" + callee + "()' is discarded — error may be silently ignored", LINT_EMPTY_CATCH)
          }
        }
      }
    }
    // Recurse
    if tag == ST_IF {
      body_arr = load64(stmt + 24)
      if body_arr != 0 { lint_check_discarded_results(load64(body_arr), load64(body_arr + 8), result_fns) }
      else_arr = load64(stmt + 32)
      if else_arr != 0 { lint_check_discarded_results(load64(else_arr), load64(else_arr + 8), result_fns) }
    }
    if tag == ST_WHILE {
      body_arr = load64(stmt + 24)
      if body_arr != 0 { lint_check_discarded_results(load64(body_arr), load64(body_arr + 8), result_fns) }
    }
    if tag == ST_FOR_IN {
      body_arr = load64(stmt + 32)
      if body_arr != 0 { lint_check_discarded_results(load64(body_arr), load64(body_arr + 8), result_fns) }
    }
    i += 1
  }
  0
}

## Linter (`sans lint`)

Static analysis that catches common issues without building:

```
sans lint foo.sans                        # lint single file
sans lint compiler/                       # lint all .sans files recursively
sans lint .                               # lint current directory
sans lint --error=unused-imports foo.sans  # promote rule to error
sans lint --quiet foo.sans                # suppress warnings
```

### Rules

| Rule | Default | Description |
|---|---|---|
| `unused-imports` | warn | Imported module never referenced |
| `unreachable-code` | warn | Code after return statement |
| `empty-catch` | warn | Result value silently discarded |
| `shadowed-vars` | warn | Inner scope redeclares outer variable |
| `unnecessary-mut` | warn | Variable declared `:=` but never reassigned |

### Configuration

Configure in `sans.json`:

```json
{
  "lint": {
    "unused-imports": "error",
    "shadowed-vars": "off"
  }
}
```

Valid severities: `"error"`, `"warn"`, `"off"`. CLI flags override config.

Exit code: 0 if no errors, 1 if any error-severity diagnostics.
