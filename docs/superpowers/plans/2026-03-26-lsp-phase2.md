# Sans LSP Phase 2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add autocomplete, didChange analysis, find-all-references, rename, semantic tokens, workspace symbols, folding ranges, and code actions to the Sans LSP server.

**Architecture:** Extends the existing Phase 1 LSP (`lsp/*.sans`) with new request handlers and deeper compiler integration. Autocomplete uses the typeck symbol tables (fn_env, structs, enums, methods, locals). Semantic tokens leverage the lexer's token stream. Find-all-references and rename use text-based search with type validation. didChange analysis throttles recompilation via a debounce timer.

**Tech Stack:** Sans (self-hosted), LLVM IR codegen, JSON-RPC over stdio, LSP protocol

```sans
handle_completion(id:J params:J) I {
  uri = params_uri(params)
  pos_encoded = params_position(params)
  line = pos_encoded / 1000000
  col = pos_encoded % 1000000

  content = doc_get(uri)
  if slen(content) == 0 {
    rpc_respond(id, make_completion_list(ja()))
    return 0
  }

  // Determine if we're after a "." (method completion)
  // Find position in content and check char before cursor
  cur_line := 0
  i := 0
  len = slen(content)
  while i < len && cur_line < line {
    if load8(content + i) == 10 { cur_line += 1 }
    i += 1
  }
  pos = i + col
  is_dot := 0
  // Walk back past the partial word to find if there's a dot
  scan := pos - 1
  while scan >= i && is_ident_char(load8(content + scan)) == 1 {
    scan -= 1
  }
  if scan >= 0 && load8(content + scan) == 46 { is_dot = 1 }

  // Extract the partial word being typed
  word_start := pos
  while word_start > i && is_ident_char(load8(content + word_start - 1)) == 1 {
    word_start -= 1
  }
  prefix = if word_start < pos { content.substring(word_start, pos) } else { "" }

  // Re-analyze to get fresh symbols
  analyze_file(uri, content)

  if is_dot == 1 {
    // Method completion — find the type of the expression before the dot
    // For now, use a simple heuristic: find the identifier before the dot
    // and look it up in locals/fn_env
    ident_end := scan
    ident_start := ident_end - 1
    while ident_start >= i && is_ident_char(load8(content + ident_start)) == 1 {
      ident_start -= 1
    }
    ident_start += 1
    if ident_start < ident_end {
      var_name = content.substring(ident_start, ident_end)
      // Look up type
      type_str := ""
      if lsp_last_symbols != 0 {
        locals_map = load64(lsp_last_symbols + 40)
        if locals_map != 0 && mhas(locals_map, var_name) {
          type_str = type_to_string(mget(locals_map, var_name))
        }
      }
      if slen(type_str) > 0 {
        rpc_respond(id, get_method_completions(type_str))
        return 0
      }
    }
    // Fallback: no type info
    rpc_respond(id, make_completion_list(ja()))
  } else {
    rpc_respond(id, get_completions(prefix))
  }
  0
}
```

```sans
// ------ Semantic tokens for Sans LSP ------
import "constants"
import "lexer"
import "protocol"

// Encode semantic tokens for a document.
// Returns LSP SemanticTokens { data: int[] }
// Data format: [deltaLine, deltaStart, length, tokenType, tokenModifiers, ...]
compute_semantic_tokens(content:S) J {
  tokens = lex(content)
  data = ja()
  prev_line := 0
  prev_col := 0
  num = load64(tokens + 8)
  tok_data = load64(tokens)
  i := 0
  while i < num {
    tok = load64(tok_data + i * 8)
    kind = tok_kind(tok)
    line = tok_line(tok) - 1
    col = tok_col(tok) - 1
    val = tok_val(tok)

    // Classify token
    sem_type := -1
    tok_len := 1

    // Keywords
    if kind >= 5 && kind <= 16 { sem_type = SEM_KEYWORD }
    if kind >= 42 && kind <= 54 { sem_type = SEM_KEYWORD }

    // Literals
    if kind == 0 { sem_type = SEM_NUMBER; tok_len = slen(str(val)) }
    if kind == 1 { sem_type = SEM_NUMBER; tok_len = slen(str(val)) + 2 }
    if kind == 2 { sem_type = SEM_STRING; tok_len = slen(val) + 2 }
    if kind == 3 { sem_type = SEM_VARIABLE; tok_len = slen(val) }

    // Comments
    if kind == 70 { sem_type = SEM_COMMENT; tok_len = slen(val) }

    // Operators
    if kind >= 31 && kind <= 41 { sem_type = SEM_OPERATOR; tok_len = 1 }
    if kind >= 59 && kind <= 67 { sem_type = SEM_OPERATOR; tok_len = 2 }

    // Type names (identifier that starts with uppercase or is a known type alias)
    if kind == 3 {
      first_ch = load8(val)
      if first_ch >= 65 && first_ch <= 90 {
        sem_type = SEM_TYPE
        tok_len = slen(val)
      }
    }

    if sem_type >= 0 {
      delta_line = line - prev_line
      delta_col = if delta_line == 0 { col - prev_col } else { col }
      data.push(ji(delta_line))
      data.push(ji(delta_col))
      data.push(ji(tok_len))
      data.push(ji(sem_type))
      data.push(ji(0))
      prev_line = line
      prev_col = col
    }

    i += 1
  }

  result = jo()
  result.set("data", data)
  result
}
```

```sans
// ------ Find references and rename for Sans LSP ------
import "constants"
import "lexer"
import "analyzer"
import "protocol"

// Find all references to the word at (line, col) in content.
// Returns JSON array of Location objects.
find_references(uri:S content:S line:I col:I include_decl:B) J {
  locations = ja()
  word = word_at_position(content, line + 1, col + 1)
  if slen(word) == 0 { return locations }

  // Lex the document and find all identifier tokens matching the word
  tokens = lex(content)
  num = load64(tokens + 8)
  tok_data = load64(tokens)
  i := 0
  while i < num {
    tok = load64(tok_data + i * 8)
    kind = tok_kind(tok)
    // TK_IDENT = 3
    if kind == 3 {
      val = tok_val(tok)
      if val == word {
        tl = tok_line(tok) - 1
        tc = tok_col(tok) - 1
        loc = make_lsp_location(uri, tl, tc, tl, tc + slen(word))
        locations.push(loc)
      }
    }
    i += 1
  }

  locations
}

// Compute workspace edit for renaming a symbol.
// Returns JSON WorkspaceEdit with changes for the document.
compute_rename(uri:S content:S line:I col:I new_name:S) J {
  refs = find_references(uri, content, line, col, 1)
  if refs.len() == 0 { return jo() }

  // Build text edits (apply in reverse order to preserve positions)
  edits = ja()
  i := 0
  while i < refs.len() {
    loc = refs.get_index(i)
    rng = loc.get("range")
    edit = jo()
    edit.set("range", rng)
    edit.set("newText", js(new_name))
    edits.push(edit)
    i += 1
  }

  // WorkspaceEdit { changes: { uri: TextEdit[] } }
  changes = jo()
  changes.set(uri, edits)
  result = jo()
  result.set("changes", changes)
  result
}
```

- [ ] **Step 1: Create `lsp/folding.sans`**

```sans
// ------ Folding ranges for Sans LSP ------
import "protocol"

// Compute folding ranges from source text.
// Returns JSON array of FoldingRange objects.
compute_folding_ranges(content:S) J {
  ranges = ja()
  len = slen(content)

  // Track brace pairs with a stack
  stack = array<I>()
  line := 0
  i := 0
  in_string := 0
  in_comment := 0

  while i < len {
    ch = load8(content + i)

    // Track strings (skip braces inside strings)
    if ch == 34 && in_comment == 0 {
      in_string = if in_string == 0 { 1 } else { 0 }
    }

    // Track line comments
    if ch == 47 && i + 1 < len && load8(content + i + 1) == 47 && in_string == 0 {
      in_comment = 1
    }
    if ch == 10 { in_comment = 0 }

    if in_string == 0 && in_comment == 0 {
      if ch == 123 {
        // '{' — push line number
        stack.push(line)
      }
      if ch == 125 {
        // '}' — pop and create range
        if stack.len() > 0 {
          start_line = stack.pop()
          if line > start_line {
            r = jo()
            r.set("startLine", ji(start_line))
            r.set("endLine", ji(line))
            r.set("kind", js("region"))
            ranges.push(r)
          }
        }
      }
    }

    if ch == 10 { line += 1 }
    i += 1
  }

  ranges
}
```

## Implementation Notes

**Import order in `lsp/main.sans`:** The imports must be: `constants`, `lexer`, `parser`, `typeck`, `rpc`, `protocol`, `analyzer`, `symbols`, `completion`, `semantic_tokens`, `references`, `folding`. The compiler resolves imports relative to the source file directory, so all `lsp/*.sans` modules can import each other by name. Compiler modules (`lexer`, `parser`, `typeck`, `constants`) are resolved because the LSP build runs from the project root with the compiler directory accessible.

**Token kinds in `compiler/constants.sans`:** The semantic tokens implementation maps token kinds to LSP semantic types. Key mappings: TK_FN(5)..TK_ENUM(15) are keywords, TK_INT_LIT(0) is number, TK_STRING_LIT(2) is string, TK_IDENT(3) is variable/type. Verify these ranges match your constants.sans before implementation.

**Autocomplete scope limitation:** Phase 2 collects the LAST function's locals, not the locals at the cursor position. This means completion will show all variables from the most recently type-checked function, which is usually the function being edited. For multi-function files, this is imprecise but usable. Cursor-precise scoping would require passing cursor position into typeck, which is a Phase 3 enhancement.

**Memory management:** The LSP process calls `scope_disable()` at startup. All allocations persist for the process lifetime. For Phase 2 this is acceptable — the editor restarts the LSP process periodically. Phase 3 could add selective cleanup between analysis runs.

**Semantic token precision:** The token classification uses lexer token kinds, which don't distinguish between function calls and variable references (both are TK_IDENT). Type names are heuristically detected by checking if the first character is uppercase. This is good enough for most Sans code where types start with uppercase (Array, Map, Result, etc.) and variables start with lowercase.
