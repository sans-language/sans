# LSP Phase 2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add local variable completion, struct field/method completion on `.`, cross-file find references, expression type hover, and scope-aware rename to the Sans LSP.

**Architecture:** Extend the compiler's typeck to emit scope/binding/expression-type data during `collect_mode`, then consume it in the LSP's analyzer, completion, references, and hover modules. Cross-file references use the existing module cache reverse-dependency graph plus lazy file discovery.

**Tech Stack:** Sans language (self-hosted compiler + LSP), JSON-RPC 2.0 over stdio, LSP 3.x protocol

```sans
// ------ Scope/Binding lookups for LSP features ---------------------------------------------------------------

// Find innermost scope containing a 0-based line
find_scope_at(line:I) I {
  if lsp_last_scopes == 0 { return 0 }
  // Convert 0-based LSP line to 1-based compiler line
  cline = line + 1
  best := 0
  best_depth := -1
  i := 0
  while i < lsp_last_scopes.len() {
    entry = lsp_last_scopes.get(i)
    sline = load64(entry + 16)
    eline = load64(entry + 24)
    depth = load64(entry + 32)
    if cline >= sline && (eline == 0 || cline <= eline) && depth > best_depth {
      best = entry
      best_depth = depth
    }
    i += 1
  }
  best
}

// Find binding declared at a specific 0-based position
find_binding_at(line:I col:I) I {
  if lsp_last_bindings == 0 { return 0 }
  cline = line + 1
  ccol = col + 1
  i := 0
  while i < lsp_last_bindings.len() {
    entry = lsp_last_bindings.get(i)
    bline = load64(entry + 16)
    bcol = load64(entry + 24)
    if bline == cline && bcol == ccol { return entry }
    i += 1
  }
  0
}

// Find the binding for a given name visible at a 0-based line
find_binding_by_name(name:S line:I) I {
  if lsp_last_bindings == 0 { return 0 }
  cline = line + 1
  best := 0
  best_line := -1
  i := 0
  while i < lsp_last_bindings.len() {
    entry = lsp_last_bindings.get(i)
    bname = load64(entry)
    bline = load64(entry + 16)
    bscope = load64(entry + 32)
    if bname == name && bline <= cline && bline > best_line {
      // Check scope contains cursor line
      scope = find_scope_by_id(bscope)
      if scope != 0 {
        sline = load64(scope + 16)
        eline = load64(scope + 24)
        if cline >= sline && (eline == 0 || cline <= eline) {
          best = entry
          best_line = bline
        }
      }
    }
    i += 1
  }
  best
}

// Find scope by ID
find_scope_by_id(scope_id:I) I {
  if lsp_last_scopes == 0 { return 0 }
  i := 0
  while i < lsp_last_scopes.len() {
    entry = lsp_last_scopes.get(i)
    if load64(entry) == scope_id { return entry }
    i += 1
  }
  0
}

// Find expression type at a 0-based position
find_expr_type_at(line:I col:I) I {
  if lsp_last_expr_types == 0 { return 0 }
  cline = line + 1
  ccol = col + 1
  i := 0
  while i < lsp_last_expr_types.len() {
    entry = lsp_last_expr_types.get(i)
    eline = load64(entry)
    ecol = load64(entry + 8)
    if eline == cline && ecol == ccol { return load64(entry + 16) }
    i += 1
  }
  0
}

// Collect all bindings visible at a 0-based line (for completion)
visible_bindings_at(line:I) I {
  if lsp_last_bindings == 0 { return array<I>() }
  cline = line + 1
  result = array<I>()
  seen = M()
  // Walk all bindings, keep the deepest-scope one for each name
  i := lsp_last_bindings.len() - 1
  while i >= 0 {
    entry = lsp_last_bindings.get(i)
    bname = load64(entry)
    bline = load64(entry + 16)
    bscope = load64(entry + 32)
    if bline <= cline && mhas(seen, bname) == 0 {
      scope = find_scope_by_id(bscope)
      if scope != 0 {
        sline = load64(scope + 16)
        eline = load64(scope + 24)
        if cline >= sline && (eline == 0 || cline <= eline) {
          result.push(entry)
          mset(seen, bname, 1)
        }
      }
    }
    i -= 1
  }
  result
}
```

```sans
// Get completions after "." — resolve receiver type and return fields/methods
get_dot_completions(content:S line:I col:I) J {
  // Find the identifier before the dot
  // col points to just after the ".", so col-1 is the dot, col-2 is end of identifier
  rcv_name = word_before_dot(content, line, col)
  if slen(rcv_name) == 0 { return make_completion_list(ja()) }

  // Look up receiver type from bindings
  binding = find_binding_by_name(rcv_name, line)
  rcv_type := 0
  if binding != 0 {
    rcv_type = load64(binding + 8)
  }

  // If not found in bindings, check fn_env (could be a module name or function)
  if rcv_type == 0 && lsp_last_symbols != 0 {
    fn_env = load64(lsp_last_symbols + 16)
    if mhas(fn_env, rcv_name) {
      sig = mget(fn_env, rcv_name)
      rcv_type = load64(sig + 8)
    }
  }

  items = ja()
  if rcv_type == 0 {
    // Unknown type — fall back to existing method completions with the name as type
    return make_completion_list(get_method_completions(rcv_name).get("items"))
  }

  // Get type name for method lookup
  type_name = type_to_string(rcv_type)

  // Check if it's a struct — add fields
  if lsp_last_symbols != 0 {
    structs_map = load64(lsp_last_symbols)
    if mhas(structs_map, type_name) {
      fields = mget(structs_map, type_name)
      fkeys = fields.keys()
      fi := 0
      while fi < fkeys.len() {
        fname = fkeys.get(fi)
        ftype = mget(fields, fname)
        item = make_completion_item(fname, 5, type_to_string(ftype))  // 5 = Field
        items.push(item)
        fi += 1
      }
    }
  }

  // Add methods for the type (from hardcoded list + methods_map)
  method_result = get_method_completions(type_name)
  method_items = method_result.get("items")
  mi := 0
  mlen = method_items.len()
  while mi < mlen {
    items.push(method_items.get(mi))
    mi += 1
  }

  make_completion_list(items)
}

// Extract identifier before the dot at given position
word_before_dot(content:S line:I col:I) S {
  // Find the line in content
  lines_arr = content.split("\n")
  if line >= lines_arr.len() { return "" }
  ln = lines_arr.get(line)
  ln_len = slen(ln)
  // col is after the dot, so dot is at col-1, identifier ends at col-2
  end := col - 2
  if end < 0 { return "" }
  // Skip any trailing spaces
  while end >= 0 && char_at(ln, end) == 32 { end -= 1 }
  if end < 0 { return "" }
  // Walk backward to find start of identifier
  start := end
  while start > 0 {
    ch = char_at(ln, start - 1)
    if (ch >= 97 && ch <= 122) || (ch >= 65 && ch <= 90) || (ch >= 48 && ch <= 57) || ch == 95 {
      start -= 1
    } else {
      start = 0 - 1  // break
    }
  }
  if start < 0 { start = 0 }
  ln.substring(start, end + 1)
}
```

```sans
// ------ Cross-file find references ---------------------------------------------------------------

g lsp_token_cache = 0

// Find references across all files that import the current module
find_references_cross_file(uri:S content:S line:I col:I) J {
  // First get local references
  local_refs = find_references(uri, content, line, col)

  // Get the word we're searching for
  word = word_at_position(content, line, col)
  if slen(word) == 0 { return local_refs }

  // Check if this is a module-level symbol (function/struct/enum)
  is_module_sym := 0
  if lsp_last_symbols != 0 {
    fn_env = load64(lsp_last_symbols + 16)
    structs = load64(lsp_last_symbols)
    enums = load64(lsp_last_symbols + 8)
    if mhas(fn_env, word) == 1 { is_module_sym = 1 }
    if mhas(structs, word) == 1 { is_module_sym = 1 }
    if mhas(enums, word) == 1 { is_module_sym = 1 }
  }

  if is_module_sym == 0 { return local_refs }

  // Get files that import this module via reverse deps
  file_path = uri_to_path(uri)
  rdeps = mcache_get_rdeps(file_path)
  if rdeps == 0 { return local_refs }

  rkeys = rdeps.keys()
  ri := 0
  while ri < rkeys.len() {
    dep_path = rkeys.get(ri)
    dep_uri = path_to_uri(dep_path)
    dep_content = doc_get(dep_uri)
    // If file not open in editor, read from disk
    if slen(dep_content) == 0 {
      dep_content = fr(dep_path)
    }
    if slen(dep_content) > 0 {
      dep_index = build_ident_index(dep_uri, dep_content)
      if mhas(dep_index, word) {
        dep_refs = mget(dep_index, word)
        // Merge into local_refs
        di := 0
        while di < dep_refs.len() {
          local_refs.push(dep_refs.get(di))
          di += 1
        }
      }
    }
    ri += 1
  }
  local_refs
}

// Get reverse deps from module cache
mcache_get_rdeps(path:S) I {
  if mcache_rdeps == 0 { return 0 }
  if mhas(mcache_rdeps, path) { return mget(mcache_rdeps, path) }
  0
}

// Convert file path to file:// URI
path_to_uri(path:S) S {
  "file://" + path
}
```

```sans
// Scope-aware rename: only renames within the binding's scope
compute_rename(uri:S content:S line:I col:I new_name:S) J {
  word = word_at_position(content, line, col)
  if slen(word) == 0 { return jo() }

  // Check if this is a local/param binding
  binding = find_binding_by_name(word, line)

  if binding != 0 {
    bkind = load64(binding + 40)
    bscope_id = load64(binding + 32)
    if bkind == 0 || bkind == 1 {
      // Local or param — rename only within scope
      return compute_scoped_rename(uri, content, word, new_name, bscope_id)
    }
  }

  // Module-level symbol — rename all occurrences in current file
  // Plus cross-file if it's exported
  edits = ja()
  index = build_ident_index(uri, content)
  if mhas(index, word) {
    refs = mget(index, word)
    ri := 0
    while ri < refs.len() {
      loc = refs.get(ri)
      edit = jo()
      edit.set("range", loc.get("range"))
      edit.set("newText", js(new_name))
      edits.push(edit)
      ri += 1
    }
  }

  // Cross-file rename for module-level symbols
  if lsp_last_symbols != 0 {
    fn_env = load64(lsp_last_symbols + 16)
    structs = load64(lsp_last_symbols)
    enums = load64(lsp_last_symbols + 8)
    is_module_sym := 0
    if mhas(fn_env, word) == 1 { is_module_sym = 1 }
    if mhas(structs, word) == 1 { is_module_sym = 1 }
    if mhas(enums, word) == 1 { is_module_sym = 1 }
    if is_module_sym == 1 {
      file_path = uri_to_path(uri)
      rdeps = mcache_get_rdeps(file_path)
      if rdeps != 0 {
        rkeys = rdeps.keys()
        ri := 0
        while ri < rkeys.len() {
          dep_path = rkeys.get(ri)
          dep_uri = path_to_uri(dep_path)
          dep_content = doc_get(dep_uri)
          if slen(dep_content) == 0 { dep_content = fr(dep_path) }
          if slen(dep_content) > 0 {
            dep_edits = compute_file_rename(dep_uri, dep_content, word, new_name)
            // Merge edits — we need a changes map keyed by URI
            // This is handled in the workspace edit builder below
          }
          ri += 1
        }
      }
    }
  }

  result = jo()
  changes = jo()
  changes.set(uri, edits)
  result.set("changes", changes)
  result
}

// Rename only within a specific scope
compute_scoped_rename(uri:S content:S word:S new_name:S scope_id:I) J {
  scope = find_scope_by_id(scope_id)
  if scope == 0 { return jo() }
  scope_start = load64(scope + 16) - 1  // convert to 0-based
  scope_end = load64(scope + 24) - 1
  if scope_end <= 0 { scope_end = 999999 }  // unclosed scope — go to end

  edits = ja()
  index = build_ident_index(uri, content)
  if mhas(index, word) {
    refs = mget(index, word)
    ri := 0
    while ri < refs.len() {
      loc = refs.get(ri)
      range = loc.get("range")
      ref_line = range.get("start").get("line").get_int()
      // Only include if within scope range
      if ref_line >= scope_start && ref_line <= scope_end {
        // Check this isn't shadowed by a deeper binding
        if is_shadowed_at(word, ref_line, scope_id) == 0 {
          edit = jo()
          edit.set("range", range)
          edit.set("newText", js(new_name))
          edits.push(edit)
        }
      }
      ri += 1
    }
  }

  result = jo()
  changes = jo()
  changes.set(uri, edits)
  result.set("changes", changes)
  result
}

// Check if a name is shadowed by a deeper binding at a given line
is_shadowed_at(name:S line:I outer_scope_id:I) I {
  if lsp_last_bindings == 0 { return 0 }
  cline = line + 1
  i := 0
  while i < lsp_last_bindings.len() {
    entry = lsp_last_bindings.get(i)
    bname = load64(entry)
    bline = load64(entry + 16)
    bscope = load64(entry + 32)
    if bname == name && bscope != outer_scope_id && bline <= cline {
      // Check if this binding's scope contains the line AND is deeper than outer
      scope = find_scope_by_id(bscope)
      if scope != 0 {
        sline = load64(scope + 16)
        eline = load64(scope + 24)
        if cline >= sline && (eline == 0 || cline <= eline) {
          // This is a shadow — check it's a child of outer_scope
          parent = load64(scope + 8)
          if parent != outer_scope_id { return 1 }
        }
      }
    }
    i += 1
  }
  0
}

// Simple file-wide rename (for cross-file module-level symbols)
compute_file_rename(uri:S content:S word:S new_name:S) J {
  edits = ja()
  index = build_ident_index(uri, content)
  if mhas(index, word) {
    refs = mget(index, word)
    ri := 0
    while ri < refs.len() {
      loc = refs.get(ri)
      edit = jo()
      edit.set("range", loc.get("range"))
      edit.set("newText", js(new_name))
      edits.push(edit)
      ri += 1
    }
  }
  edits
}
```
