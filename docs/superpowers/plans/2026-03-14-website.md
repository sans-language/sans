**Tech Stack:** Rust (compiler), C (runtime), LLVM (codegen), Sans (website), HTML/CSS (static content)

```c
long cy_string_ends_with(const char* s, const char* suffix) {
    long suffix_len = (long)strlen(suffix);
    long s_len = (long)strlen(s);
    if (suffix_len > s_len) return 0;
    return memcmp(s + s_len - suffix_len, suffix, suffix_len) == 0 ? 1 : 0;
}
```

```c
long cy_http_respond(CyHttpRequest* req, long status, const char* body, const char* content_type) {
    if (req->client_fd < 0) return 0;

    const char* status_text = "OK";
    if (status == 404) status_text = "Not Found";
    else if (status == 500) status_text = "Internal Server Error";
    else if (status == 400) status_text = "Bad Request";

    if (content_type == NULL) {
        content_type = "text/html; charset=utf-8";
    }

    long body_len = (long)strlen(body);
    char header[1024];
    int header_len = snprintf(header, sizeof(header),
        "HTTP/1.1 %ld %s\r\n"
        "Content-Length: %ld\r\n"
        "Content-Type: %s\r\n"
        "Connection: close\r\n"
        "\r\n",
        status, status_text, body_len, content_type);
```

The rest of the function (send loop, close, free) stays the same.

### Task 6: Create website main.sans

**Files:**
- Create: `website/main.sans`

- [ ] **Step 1: Write main.sans**

```sans
mime(path:S) S {
  path.ew(".html") ? "text/html; charset=utf-8" : path.ew(".css") ? "text/css" : path.ew(".js") ? "application/javascript" : path.ew(".png") ? "image/png" : path.ew(".ico") ? "image/x-icon" : "text/plain"
}

serve(req:HttpRequest, fpath:S) I {
  file_exists(fpath) ? req.respond(200, file_read(fpath), mime(fpath)) : req.respond(404, "Not Found")
}

main() {
  p("Sans docs server: http://localhost:8080")
  server = listen(8080)
  while true {
    req = server.accept
    path = req.path
    path.contains("..") ? req.respond(400, "Bad Request") : path == "/" ? serve(req, "static/index.html") : path == "/docs" ? serve(req, "static/docs.html") : path == "/benchmarks" ? serve(req, "static/benchmarks.html") : path.sw("/static/") ? serve(req, "static/" + path.substring(8, path.len())) : req.respond(404, "Not Found")
  }
  0
}
```
